## 基本使用

useState , useEffect
```
import React, { useState, useEffect } from "react";

export default function () {
  const [aa, onChangeAA] = useState(10);



// useEffect  第二个参数 为[]时候，只会在componentDidMount中执行
  useEffect(() => {
    // 更新文档的标题
    document.title = `你点击了${aa} 次`;
  }, [aa])



  return (
    <div>
      <button onClick={() => {
        onChangeAA(aa + 1)
      }}>+</button>
      {aa}
      <button onClick={() => {
        onChangeAA(aa - 1)
      }}>-</button>
    </div>
  )
}  

```
